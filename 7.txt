<script>
var clickCount = 0;
var isDay = true;
var trees = document.getElementsByClassName("tree");
var scene = document.getElementById("scene");
var sun = document.getElementById("sun");
var chirp = document.getElementById("chirp");
var sw = document.getElementById("switch");
var soundStarted = false;
function startSound(){
    if(!soundStarted){
        chirp.volume = 0.3;
        chirp.play();
        soundStarted = true;
    }
}
function pressMe(){
    startSound();
    clickCount++;
    if(isDay){
        for(var i=0;i<trees.length;i++){
            trees[i].style.transform += " scaleY(1.08)";
        }
    }
    if(!isDay){
        for(var i=0;i<trees.length;i++){
            trees[i].style.transform += " scaleY(0.92)";
        }
    }
    if(clickCount >= 5){
        clickCount = 0;

        if(isDay){
            toNight();
        }
        else{
            toDay();
        }

        isDay = !isDay;
    }
}
function toNight(){
    scene.style.background = "#0b0b2b";
    sun.style.top = "90%";
    sw.play();
}
function toDay(){
    scene.style.background = "skyblue";
    sun.style.top = "60px";
    sw.play();
}
setInterval(moveBirds,1200);
function moveBirds(){
    for(var i=1;i<=4;i++){
        var b = document.getElementById("b"+i);
        var l = Math.random()*90;
        var t = Math.random()*60;
        b.style.left = l + "%";
        b.style.top = t + "%";
    }
}
