<input type="text" id="dateInput" placeholder="10/11/2009">
<button onclick="showDay()">Get Day</button>
<p id="result"></p>
<script>
function getDayName(dateString) {
    let days = ["Sunday", "Monday", "Tuesday", "Wednesday",
                "Thursday", "Friday", "Saturday"];

    let date = new Date(dateString);
    return days[date.getDay()];
}
function showDay() {
    let input = document.getElementById("dateInput").value;
    let day = getDayName(input);

    document.getElementById("result").innerHTML =
        "Day is: " + day;
}
</script>

<input type="text" id="textInput">

function regexVar() {
    return /^([aeiou]).*\1$/i;
}
function check() {
    let s = document.getElementById("textInput").value;
    let re = regexVar();
    document.getElementById("result").innerHTML =
        re.test(s);
}

<p id="result"></p>
<script>
// Function to modify array
function modifyArray(arr) {
    var modified = [];
    for (var i = 0; i < arr.length; i++) {
        if (arr[i] % 2 === 0) {
            modified.push(arr[i] * 2);
        } else {
            modified.push(arr[i] * 3);
        }
    }
    return modified;
}
document.getElementById("processBtn").onclick = function() {
    var input = document.getElementById("arrayInput").value;
    var arr = [];
    var parts = input.split(" ");
    for (var i = 0; i < parts.length; i++) {
        var num = Number(parts[i]);
        if (isNaN(num)) {
            document.getElementById("result").innerHTML = "Please enter valid numbers!";
            return;
        }
        arr.push(num);
    }
    var modified = modifyArray(arr);
    document.getElementById("result").innerHTML = "Modified Array: " + modified.join(" ");
};
</script>
</body>
</html>
