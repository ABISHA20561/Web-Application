7.<!DOCTYPE html>
<html>
<head>
    <title>Vowels and Consonants</title>
</head>
<body>
    <h2>Check Vowels and Consonants</h2>
    <input type="text" id="inputString" placeholder="Enter a string">
    <button onclick="vowelsAndConsonants()">Submit</button>
    <h3>Output:</h3>
    <pre id="output"></pre>
    <script>
        function vowelsAndConsonants() {
            const s = document.getElementById('inputString').value;
            let vowelOutput = '';
            let consonantOutput = '';
            for (let i = 0; i < s.length; i++) {
                let char = s[i];
                if (char === 'a' || char === 'e' || char === 'i' || char === 'o' || char === 'u') {
                    vowelOutput += char + '\n';
                } else {
                    consonantOutput += char + '\n';
                }
            }
            document.getElementById('output').textContent = vowelOutput + consonantOutput;
        }
    </script>
</body>

8.<!DOCTYPE html>
<html>
<head>
    <title>Switch check</title>
</head>
<body>
    <h2>Get Letter Based on First Character</h2>
    <input type="text" id="inputString" placeholder="Enter a string (lowercase only)">
    <button onclick="showLetter()">Submit</button>
    <h3>Output:</h3>
    <pre id="output"></pre>
    <script>
        function getLetter(s) {
            let firstChar = s[0];
            let letter;
            switch(true) {
                case firstChar === 'a' || firstChar === 'e' || firstChar === 'i' || firstChar === 'o' || firstChar === 'u':
                    letter = 'A';
                    break;
                case firstChar === 'b' || firstChar === 'c' || firstChar === 'd' || firstChar === 'f' || firstChar === 'g':
                    letter = 'B';
                    break;
                case firstChar === 'h' || firstChar === 'j' || firstChar === 'k' || firstChar === 'l' || firstChar === 'm':
                    letter = 'C';
                    break;
                case firstChar === 'n' || firstChar === 'p' || firstChar === 'q' || firstChar === 'r' || firstChar === 's' ||
                     firstChar === 't' || firstChar === 'v' || firstChar === 'w' || firstChar === 'x' || firstChar === 'y' || firstChar === 'z':
                    letter = 'D';
                    break;
                default:
                    letter = 'Invalid character';
            }

            return letter;
        }

        function showLetter() {
            const s = document.getElementById('inputString').value.trim().toLowerCase();
            if (s.length === 0) {
                document.getElementById('output').textContent = "Please enter a string!";
                return;
            }
            const result = getLetter(s);
            document.getElementById('output').textContent = result;
        }
    </script>
</body>
</html>

<script>
function getGrade(score) {
    if (score > 25 && score <= 30) return "A";
    else if (score > 20 && score <= 25) return "B";
    else if (score > 15 && score <= 20) return "C";
    else if (score > 10 && score <= 15) return "D";
    else if (score > 5 && score <= 10) return "E";
    else if (score >= 0 && score <= 5) return "F";
    else return "Invalid score";
}
document.getElementById("checkBtn").onclick = function() {
    var score = parseInt(document.getElementById("scoreInput").value);
    var grade = getGrade(score);
    document.getElementById("result").innerHTML = "Grade: " + grade;
};
</script>

10.<!DOCTYPE html>
<html>
<head>
    <title>Factorial Calculator</title>
</head>
<body>
    <h1>Factorial Calculator</h1>
    <input type="number" id="number">
    <button onclick="calculateFactorial()">Calculate</button>
    <p id="result"></p>
    <script>
        function factorial(n) {
            let result = 1;
            for (let i = 2; i <= n; i++) {
                result *= i;
            }
            return result;
        }
       function calculateFactorial() {
            let num = parseInt(document.getElementById('number').value);
            if (num < 1 || num > 10 || isNaN(num)) {
                document.getElementById('result').innerText = "Please enter a number between 1 and 10.";
                return;
            }
            let fact = factorial(num);
            document.getElementById('result').innerText = num + "! = " + fact;
        }
    </script>
</body>
</html>
