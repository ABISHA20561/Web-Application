        <h2>Action Menu</h2>
        <input type="text" id="userInput" placeholder="Type something here...">
        <div class="btn-group">
            <button onclick="openNewWindow()">1. Open Window</button>
            <button onclick="closeNewWindow()">2. Close Window</button>
            <button onclick="convertToUppercase()">3. Convert to Uppercase</button>
        </div>
        <div id="output">Result will appear here...</div>
    </div>
    <script>
        let myWindow;
        function openNewWindow() {
            myWindow = window.open("", "NewWindow", "width=400,height=300");
            myWindow.document.write("<h2>Window Active</h2><p>Opened via Button Click</p>");
            document.getElementById('output').innerText = "Status: New window opened.";
        }
        function closeNewWindow() {
            if (myWindow && !myWindow.closed) {
                myWindow.close();
                document.getElementById('output').innerText = "Status: Window closed.";
            } else {
                document.getElementById('output').innerText = "Error: No active window to close.";
            }
        }
        function convertToUppercase() {
            const inputVal = document.getElementById('userInput').value;
            const outputDiv = document.getElementById('output');
            if (inputVal.trim() === "") {
                outputDiv.innerText = "Error: Please enter some text first!";
            } else {
                outputDiv.innerText = "Uppercase: " + inputVal.toUpperCase();
            }
        }
    </script>
